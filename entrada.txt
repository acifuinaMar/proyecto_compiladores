program {
    int x = 10;
    string prefijo = "El resultado es: ";
    int factorial(int n) {
        si (n <= 1) {
            return 1;
        } sino {
            return n * factorial(n - 1);
        }
    }
    while (x > 0) {
        int temp = factorial(x);
        print(prefijo);
        print(temp);
        x = x - 2;
    }
}
